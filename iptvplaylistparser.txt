  # Xumo Playlist Generator Service
  xumo-playlist-generator:
    container_name: xumo-playlist-generator
    image: python:3.11-slim
    command: >
      sh -c "
      apt-get update &&
      apt-get install -y curl gzip &&
      mkdir -p /app/playlists /app/epg &&
      while true; do
        echo 'Downloading xumo_playlist.m3u...' &&
        curl -L -o /app/playlists/xumo_playlist.m3u 'https://raw.githubusercontent.com/BuddyChewChew/xumo-playlist-generator/refs/heads/main/playlists/xumo_playlist.m3u' &&
        echo 'Downloading EPG file...' &&
        curl -L -o /app/epg/xumo_epg.xml.gz 'https://github.com/BuddyChewChew/xumo-playlist-generator/raw/refs/heads/main/playlists/xumo_epg.xml.gz' &&
        echo 'Extracting EPG file...' &&
        gunzip -f /app/epg/xumo_epg.xml.gz &&
        echo 'Files ready - Next update in 6 hours' &&
        sleep 21600;
      done & python -m http.server 8080 --directory /app
      "
    environment:
      - TZ=America/Chicago
      - PYTHONUNBUFFERED=1
    networks:
      - xumo-network
    ports:
      - "9005:8080"
    restart: unless-stopped
    volumes:
      - /mnt/System/APP_Configs/IPTV_Playlists/xumotv/playlists:/app/playlists
      - /mnt/System/APP_Configs/IPTV_Playlists/xumotv/epg:/app/epg
    working_dir: /app
    deploy:
      resources:
        limits:
          cpus: '1'
          memory: 512M
        reservations:
          cpus: '0.25'
          memory: 256M
    logging:
      driver: json-file
      options:
        max-file: '3'
        max-size: 10m
